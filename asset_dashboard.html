STEP 1 — Google Apps Script を作成
1. script.google.com を開く
2. 「新しいプロジェクト」→ 下のコードを貼り付け → 1行目のSECRETを自分の合言葉に変更 → 保存（Ctrl+S）
3. 「デプロイ」→「新しいデプロイ」→ 種類:ウェブアプリ → アクセス:全員 → デプロイ
4. 表示された ウェブアプリURL をコピーして下に貼り付け 
const SECRET = 'ここを自分の合言葉に変える'; // 例: 'haneda2026fire'

function doPost(e) {
  if (e.parameter.key !== SECRET) {
    return ContentService.createTextOutput('unauthorized');
  }
  var data = e.postData.contents;
  var fileId = PropertiesService.getScriptProperties().getProperty('FILE_ID');
  if (!fileId) {
    var file = DriveApp.createFile('asset_history.json', data, 'application/json');
    PropertiesService.getScriptProperties().setProperty('FILE_ID', file.getId());
  } else {
    DriveApp.getFileById(fileId).setContent(data);
  }
  return ContentService.createTextOutput('ok');
}
function doGet(e) {
  if (e.parameter.key !== SECRET) {
    return ContentService.createTextOutput('unauthorized');
  }
  var fileId = PropertiesService.getScriptProperties().getProperty('FILE_ID');
  if (!fileId) return ContentService.createTextOutput('[]');
  return ContentService.createTextOutput(
    DriveApp.getFileById(fileId).getBlob().getDataAsString()
  );
}
